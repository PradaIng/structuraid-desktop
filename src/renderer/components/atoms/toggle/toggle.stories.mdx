import { Canvas, Meta, Story } from '@storybook/addon-docs';
import { COLORS } from '../../../constants'

import Toggle from './toggle';

<Meta title="Atoms/Toggle" component={Toggle} />

export const Template = (args) => {
  return (
    <div style={{
      padding: '30px',
      background: args.color === 'dark' ? COLORS.PRIMARY : COLORS.BLUE_WHITE
    }}>
      <Toggle {...args} />
    </div>
  )
};

# Toggle Field

##  Usage

This is the `Toggle` atom.

<Canvas>
  <Story
    name="default"
    args={{}}
  >
    {Template.bind({})}
  </Story>
</Canvas>

Use the `checked` and `onChange` props to control it

<Canvas>
  <Story
    name="controlled"
    args={{
      checked: true,
      onChange: (value) => alert(value)
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>

Use the `disabled` prop to make it unusable.

<Canvas>
  <Story
    name="disabled"
    args={{
      disabled: true
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
